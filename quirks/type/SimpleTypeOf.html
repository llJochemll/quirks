    Alias SimpleTypeOf

    Returns the same as TypeOf, but but does away with pointers

    alias SimpleTypeOf(alias thing) = SimpleTypeOf!(std.traits.PointerTarget!Type);

    Example

    struct S {
    long id;
    int age;
    string name() {
        return "name";
    }
}
int number;
auto s = new S;

TypeOf!int; // int
TypeOf!number; // int
TypeOf!(S**); // S
TypeOf!s; // S
